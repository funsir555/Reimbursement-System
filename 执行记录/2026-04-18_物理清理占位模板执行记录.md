## 正式执行 SQL 口径

```sql
START TRANSACTION;

CREATE TEMPORARY TABLE tmp_template_cleanup AS
SELECT id, template_code
FROM pm_document_template
WHERE id NOT IN (18,22,23,24);

CREATE TEMPORARY TABLE tmp_document_cleanup AS
SELECT DISTINCT document_code
FROM pm_document_instance
WHERE template_code IN (SELECT template_code FROM tmp_template_cleanup);

DELETE l
FROM pm_document_action_log l
JOIN tmp_document_cleanup d ON l.document_code = d.document_code;

DELETE i
FROM pm_document_instance i
JOIN tmp_document_cleanup d ON i.document_code = d.document_code;

DELETE s
FROM pm_template_scope s
JOIN tmp_template_cleanup t ON s.template_id = t.id;

DELETE p
FROM pm_document_template p
JOIN tmp_template_cleanup t ON p.id = t.id;

COMMIT;
```
